62 ~GenericQueue() {
63 UASSERT(consumers_count_ == kCreatedAndDead || !consumers_count_);
64 UASSERT(producers_count_ == kCreatedAndDead || !producers_count_);
65
66 // Clear remaining items in queue
67 T value;
68 ConsumerToken token{queue_};
69 while (consumer_side_.PopNoblock(token, value)) {
70 }
71
72 // Return value of semaphores to start state.
73 if (producers_count_ == kCreatedAndDead) {
74 consumer_side_.ResumeBlockingOnPop();
75 }
76 if (consumers_count_ == kCreatedAndDead) {
77 producer_side_.DecreaseCapacity(kSemaphoreUnlockValue);
78 }
79 SetSoftMaxSize(0);
80 }

94 Producer GetProducer() {
95 std::size_t old_producers_count{};
96 utils::AtomicUpdate(producers_count_, [&](auto old_value) {
97 old_producers_count = old_value;
98 return old_value == kCreatedAndDead ? 1 : old_value + 1;
99 });
100
101 if (old_producers_count == kCreatedAndDead) {
102 consumer_side_.ResumeBlockingOnPop();
103 }
104 UASSERT(MultipleProducer || old_producers_count != 1);
105
106 return Producer(this->shared_from_this(), EmplaceEnabler{});
107 }

110 Consumer GetConsumer() {
111 std::size_t old_consumers_count{};
112 utils::AtomicUpdate(consumers_count_, [&](auto old_value) {
113 old_consumers_count = old_value;
114 return old_value == kCreatedAndDead ? 1 : old_value + 1;
115 });
116
117 if (old_consumers_count == kCreatedAndDead) {
118 producer_side_.DecreaseCapacity(kSemaphoreUnlockValue);
119 }
120 UASSERT(MultipleConsumer || old_consumers_count != 1);
121 return Consumer(this->shared_from_this(), EmplaceEnabler{});
122 }

128 void SetSoftMaxSize(std::size_t max_size) {
129 max_size = std::min(max_size, kUnbounded);
130 const auto old_capacity = capacity_.exchange(max_size);
131 if (max_size > old_capacity) {
132 producer_side_.IncreaseCapacity(max_size - old_capacity);
133 } else if (max_size < old_capacity) {
134 producer_side_.DecreaseCapacity(old_capacity - max_size);
135 }
136 }

178 void MarkConsumerIsDead() {
179 const auto new_consumers_count =
180 utils::AtomicUpdate(consumers_count_, [](auto old_value) {
181 return old_value == 1 ? kCreatedAndDead : old_value - 1;
182 });
183 if (new_consumers_count == kCreatedAndDead) {
184 producer_side_.IncreaseCapacity(kSemaphoreUnlockValue);
185 }
186 }
187
188 void MarkProducerIsDead() {
189 const auto new_producers_count =
190 utils::AtomicUpdate(producers_count_, [](auto old_value) {
191 return old_value == 1 ? kCreatedAndDead : old_value - 1;
192 });
193 if (new_producers_count == kCreatedAndDead) {
194 consumer_side_.StopBlockingOnPop();
195 }
196 }
197
198 bool NoMoreConsumers() const { return consumers_count_ == kCreatedAndDead; }
199
200 bool NoMoreProducers() const { return producers_count_ == kCreatedAndDead; }

202 void DoPush(ProducerToken& token, T&& value) {
203 if constexpr (MultipleProducer) {
204 queue_.enqueue(token, std::move(value));
205 } else {
206 // If producer is single substitute his token
207 queue_.enqueue(single_producer_token_, std::move(value));
208 }
209
210 consumer_side_.OnElementPushed();
211 }
212
213 [[nodiscard]] bool DoPop(ConsumerToken& token, T& value) {
214 bool success = false;
215 if constexpr (MultipleProducer) {
216 success = queue_.try_dequeue(token, value);
217 } else {
218 // Substitute with our single producer token
219 success = queue_.try_dequeue_from_producer(single_producer_token_, value);
220 }
221
222 if (success) {
223 producer_side_.OnElementPopped();
224 return true;
225 }
226
227 return false;
228 }
229
230 private:
231 moodycamel::ConcurrentQueue<T> queue_{1};
232 std::atomic<std::size_t> consumers_count_{0};
233 std::atomic<std::size_t> producers_count_{0};
234 std::atomic<std::size_t> capacity_;

236 SingleProducerToken single_producer_token_;

242 std::numeric_limits<std::size_t>::max();
243 static constexpr std::size_t kSemaphoreUnlockValue =
244 std::numeric_limits<std::size_t>::max() / 2;

248template <typename T, bool MP, bool MC>
249class GenericQueue<T, MP, MC>::SingleProducerSide final {
250 public:
251 explicit SingleProducerSide(GenericQueue& queue)
252 : queue_(queue), remaining_capacity_(0) {}
253
256 [[nodiscard]] bool Push(ProducerToken& token, T&& value,
257 engine::Deadline deadline) {
258 if (DoPush(token, std::move(value))) {
259 return true;
260 }
261
262 return nonfull_event_.WaitForEventUntil(deadline) &&
263 DoPush(token, std::move(value));
264 }
265
266 [[nodiscard]] bool PushNoblock(ProducerToken& token, T&& value) {
267 return DoPush(token, std::move(value));
268 }
269
270 void OnElementPopped() {
271 ++remaining_capacity_;
272 nonfull_event_.Send();
273 }
274
275 void DecreaseCapacity(std::size_t count) { remaining_capacity_ -= count; }
276
277 void IncreaseCapacity(std::size_t count) {
278 remaining_capacity_ += count;
279 nonfull_event_.Send();
280 }
281
282 private:
283 [[nodiscard]] bool DoPush(ProducerToken& token, T&& value) {
284 if (queue_.NoMoreConsumers() || !remaining_capacity_) {
285 return false;
286 }
287
288 --remaining_capacity_;
289 queue_.DoPush(token, std::move(value));
290 nonfull_event_.Reset();
291 return true;
292 }
293
294 GenericQueue& queue_;
295 engine::SingleConsumerEvent nonfull_event_;
296 std::atomic<std::size_t> remaining_capacity_;
297};

302 public:
303 explicit MultiProducerSide(GenericQueue& queue)

321 }

324 return remaining_capacity_.try_lock_shared() &&
325 DoPush(token, std::move(value));

329
330 void DecreaseCapacity(std::size_t count) {
331 remaining_capacity_.lock_shared_count(count);

335 remaining_capacity_.unlock_shared_count(count);

340 if (queue_.NoMoreConsumers()) {

354template <typename T, bool MP, bool MC>
355class GenericQueue<T, MP, MC>::SingleConsumerSide final {
356 public:
357 explicit SingleConsumerSide(GenericQueue& queue) : queue_(queue), size_(0) {}
358
360 [[nodiscard]] bool Pop(ConsumerToken& token, T& value,
361 engine::Deadline deadline) {
362 while (!DoPop(token, value)) {
363 if (queue_.NoMoreProducers() ||
364 !nonempty_event_.WaitForEventUntil(deadline)) {
365 // Producer might have pushed something in queue between .pop()
366 // and !producer_is_created_and_dead_ check. Check twice to avoid
367 // TOCTOU.
368 return DoPop(token, value);
369 }
370 }
371 return true;
372 }
373
374 [[nodiscard]] bool PopNoblock(ConsumerToken& token, T& value) {
375 return DoPop(token, value);
376 }
377
378 void OnElementPushed() {
379 ++size_;
380 nonempty_event_.Send();
381 }
382
383 void StopBlockingOnPop() { nonempty_event_.Send(); }
384
385 void ResumeBlockingOnPop() {}
386
387 std::size_t GetSize() const { return size_; }
388
389 private:
390 [[nodiscard]] bool DoPop(ConsumerToken& token, T& value) {
391 if (queue_.DoPop(token, value)) {
392 --size_;
393 nonempty_event_.Reset();
394 return true;
395 }
396 return false;
397 }
398
399 GenericQueue& queue_;
400 engine::SingleConsumerEvent nonempty_event_;
401 std::atomic<std::size_t> size_;
402};

405template <typename T, bool MP, bool MC>
406class GenericQueue<T, MP, MC>::MultiConsumerSide final {
407 public:
408 explicit MultiConsumerSide(GenericQueue& queue)
409 : queue_(queue), size_(kSemaphoreUnlockValue) {
410 const bool success = size_.try_lock_shared_count(kSemaphoreUnlockValue);
411 UASSERT(success);
412 }
413
414 ~MultiConsumerSide() { size_.unlock_shared_count(kSemaphoreUnlockValue); }
415
417 [[nodiscard]] bool Pop(ConsumerToken& token, T& value,
418 engine::Deadline deadline) {
419 return size_.try_lock_shared_until(deadline) && DoPop(token, value);
420 }
421
422 [[nodiscard]] bool PopNoblock(ConsumerToken& token, T& value) {
423 return size_.try_lock_shared() && DoPop(token, value);
424 }
425
426 void OnElementPushed() { size_.unlock_shared(); }
427
428 void StopBlockingOnPop() { size_.unlock_shared_count(kSemaphoreUnlockValue); }
429
430 void ResumeBlockingOnPop() { size_.lock_shared_count(kSemaphoreUnlockValue); }
431
432 std::size_t GetSize() const {
433 std::size_t cur_size = size_.RemainingApprox();
434 if (cur_size < kUnbounded) {
435 return cur_size;
436 } else if (cur_size <= kSemaphoreUnlockValue) {
437 return 0;
438 }
439 return cur_size - kSemaphoreUnlockValue;
440 }
441
442 private:
443 [[nodiscard]] bool DoPop(ConsumerToken& token, T& value) {
444 if (!queue_.DoPop(token, value)) {
445 size_.unlock_shared();
446 return false;
447 }
448 return true;
449 }
450
451 GenericQueue& queue_;
452 engine::Semaphore size_;
453};
