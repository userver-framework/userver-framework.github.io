115/// @brief Compute (a - b) with a specified duration
116template <class Duration, class Clock>
117double CalcTimeDiff(const std::chrono::time_point<Clock>& a,
118 const std::chrono::time_point<Clock>& b) {
119 const auto duration_a = a.time_since_epoch();
120 const auto duration_b = b.time_since_epoch();
121 return std::chrono::duration_cast<Duration>(duration_a - duration_b).count();
122}
