708template <typename T>
709void Row::To(T&& val, RowTag) const {
710 // Convert the val into a writable tuple and extract the data
711 using ValueType = std::decay_t<T>;
712 static_assert(io::traits::kIsRowType<ValueType>,
713 "This type cannot be used as a row type");
714 using RowType = io::RowType<ValueType>;
715 using TupleType = typename RowType::TupleType;
716 constexpr auto tuple_size = RowType::size;
717 if (tuple_size > Size()) {
718 throw InvalidTupleSizeRequested(Size(), tuple_size);
719 } else if (tuple_size < Size()) {
720 LOG_LIMITED_WARNING()
721 << "Row size is greater that the number of data members in "
722 "C++ user datatype "
723 << compiler::GetTypeName<T>();
724 }
725
726 detail::TupleDataExtractor<TupleType>::ExtractTuple(
727 *this, RowType::GetTuple(std::forward<T>(val)));
728}

730template <typename T>
731void Row::To(T&& val, FieldTag) const {
732 using ValueType = std::decay_t<T>;
733 // composite types can be parsed without an explicit mapping
734 static_assert(io::traits::kIsMappedToPg<ValueType> ||
735 io::traits::kIsCompositeType<ValueType>,
736 "This type is not mapped to a PostgreSQL type");
737 // Read the first field into the type
738 if (Size() < 1) {
739 throw InvalidTupleSizeRequested{Size(), 1};
740 }
741 if (Size() > 1) {
742 throw NonSingleColumResultSet{Size(), compiler::GetTypeName<T>(), "As"};
743 }
744 (*this)[0].To(std::forward<T>(val));
745}
