Detailed Description

template<class Container, class Projection>
class utils::ProjectingView< Container, Projection >

View that applies projection on iterator dereference.

Example:

std::map<int, char> cont{
{1, '1'},
{2, '2'},
};
auto proj = utils::ProjectingView(
cont, [](auto& v) -> char& { return v.second; });
assert(*proj.begin() == 1);
assert(*++proj.begin() == 2);
*proj.begin() = 'Y';
assert(*proj.begin() == 'Y');
assert(cont[1] == 'Y');
