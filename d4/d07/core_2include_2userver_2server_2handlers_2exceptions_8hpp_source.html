183 }

187 }
188
189 void Apply(formats::json::Value details_) { details = std::move(details_); }
190
191 template <typename MessageBuilder>
192 void Apply(MessageBuilder&& builder) {
193 impl::MessageExtractor<MessageBuilder> extractor{builder};
194 is_external_body_formatted = extractor.IsExternalBodyFormatted();
195 service_code = extractor.GetServiceCode();
196 external_body = extractor.GetExternalBody();
197 internal_message = extractor.GetInternalMessage();
198 }

220class CustomHandlerException : public std::runtime_error {
221 public:
222 // Type aliases for usage in descenant classes that are in other namespaces
223 using HandlerErrorCode = handlers::HandlerErrorCode;
224 using ServiceErrorCode = handlers::ServiceErrorCode;
225 using InternalMessage = handlers::InternalMessage;
226 using ExternalBody = handlers::ExternalBody;
227
228 constexpr static HandlerErrorCode kDefaultCode =
229 HandlerErrorCode::kUnknownError;
230
231 public:
232 CustomHandlerException(impl::CustomHandlerExceptionData data)
233 : runtime_error(data.internal_message.empty()
234 ? GetCodeDescription(data.handler_code)
235 : data.internal_message),
236 data_(std::move(data)) {
237 UASSERT_MSG(data_.details.IsNull() || data_.details.IsObject(),
238 "The details JSON value must be either null or an object");
239 }
240
241 // Constructor for throwing with all arguments specified
242 CustomHandlerException(ServiceErrorCode service_code,
243 ExternalBody external_body,
244 InternalMessage internal_message,
245 HandlerErrorCode handler_code,
246 formats::json::Value details = {})
247 : CustomHandlerException(impl::CustomHandlerExceptionData{
248 std::move(service_code), std::move(external_body),
249 std::move(internal_message), handler_code, std::move(details)}) {}
250
251 template <typename MessageBuilder>
252 CustomHandlerException(MessageBuilder&& builder,
253 HandlerErrorCode handler_code)
254 : CustomHandlerException(impl::CustomHandlerExceptionData{
255 std::forward<MessageBuilder>(builder), handler_code}) {}
256
257 HandlerErrorCode GetCode() const { return data_.handler_code; }
258
259 const std::string& GetServiceCode() const { return data_.service_code; }
260
261 bool IsExternalErrorBodyFormatted() const {
262 return data_.is_external_body_formatted;
263 }
264
265 const std::string& GetExternalErrorBody() const {
266 return data_.external_body;
267 }
268
269 const formats::json::Value& GetDetails() const { return data_.details; };
270
271 private:
272 const impl::CustomHandlerExceptionData data_;
273};
