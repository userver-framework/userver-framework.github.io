202
203 void Apply(formats::json::Value details_) { details = std::move(details_); }

207 impl::MessageExtractor<MessageBuilder> extractor{builder};

209 service_code = extractor.GetServiceCode();

234class CustomHandlerException : public std::runtime_error {
235 public:
236 // Type aliases for usage in descenant classes that are in other namespaces
237 using HandlerErrorCode = handlers::HandlerErrorCode;
238 using ServiceErrorCode = handlers::ServiceErrorCode;
239 using InternalMessage = handlers::InternalMessage;
240 using ExternalBody = handlers::ExternalBody;
241 using ExtraHeaders = handlers::ExtraHeaders;
242
243 constexpr static HandlerErrorCode kDefaultCode =
244 HandlerErrorCode::kUnknownError;
245
246 CustomHandlerException(impl::CustomHandlerExceptionData data)
247 : runtime_error(data.internal_message.empty()
248 ? GetCodeDescription(data.handler_code)
249 : data.internal_message),
250 data_(std::move(data)) {
251 UASSERT_MSG(data_.details.IsNull() || data_.details.IsObject(),
252 "The details JSON value must be either null or an object");
253 }
254
255 // Constructor for throwing with all arguments specified
256 CustomHandlerException(ServiceErrorCode service_code,
257 ExternalBody external_body,
258 InternalMessage internal_message,
259 HandlerErrorCode handler_code,
260 ExtraHeaders headers = {},
261 formats::json::Value details = {})
262 : CustomHandlerException(impl::CustomHandlerExceptionData{
263 std::move(service_code), std::move(external_body),
264 std::move(internal_message), handler_code, std::move(headers),
265 std::move(details)}) {}
266
267 template <typename MessageBuilder>
268 CustomHandlerException(MessageBuilder&& builder,
269 HandlerErrorCode handler_code)
270 : CustomHandlerException(impl::CustomHandlerExceptionData{
271 std::forward<MessageBuilder>(builder), handler_code}) {}

273 HandlerErrorCode GetCode() const { return data_.handler_code; }

289 private:
290 impl::CustomHandlerExceptionData data_;
291};
