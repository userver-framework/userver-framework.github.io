173template <typename... Options>
174std::optional<formats::bson::Document> Collection::FindOne(
175 formats::bson::Document filter, Options&&... options) const {
176 static_assert(
177 !(std::is_same<std::decay_t<Options>, options::Limit>::value || ...),
178 "Limit option cannot be used in FindOne");
179 auto cursor = Find(std::move(filter), options::Limit{1},
180 std::forward<Options>(options)...);
181 if (cursor.begin() == cursor.end()) return {};
182 return *cursor.begin();
183}
