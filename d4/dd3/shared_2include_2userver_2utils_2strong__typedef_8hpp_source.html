139template <class To, class... Args>
140constexpr bool IsStrongToStrongConversion() noexcept {
141 static_assert(IsStrongTypedef<To>::value);
142
143 // NOLINTNEXTLINE(bugprone-suspicious-semicolon)
144 if constexpr (sizeof...(Args) == 1) {
145 using FromDecayed = std::decay_t<decltype((std::declval<Args>(), ...))>;
146 // NOLINTNEXTLINE(bugprone-suspicious-semicolon)
147 if constexpr (IsStrongTypedef<FromDecayed>::value) {
148 // Required to make `MyVariant v{MySpecialInt{10}};` compile.
149 return !std::is_same_v<FromDecayed, To> &&
150 (std::is_same_v<typename FromDecayed::UnderlyingType,
151 typename To::UnderlyingType> ||
152 std::is_arithmetic_v<typename To::UnderlyingType>);
153 }
154 }
155
156 return false;
157}
