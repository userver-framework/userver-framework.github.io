55template <typename T>
56struct CompositeBinaryParser : BufferParserBase<T> {
57 using BaseType = BufferParserBase<T>;
58 using RowType = io::RowType<T>;
59 using IndexSequence = typename RowType::IndexSequence;
60
61 using BaseType::BaseType;
62
63 void operator()(FieldBuffer buffer, const TypeBufferCategory& categories) {
64 if constexpr (!traits::kIsMappedToUserType<T>) {
65 InitRecordParser();
66 }
67
68 Integer field_count{0};
69 buffer.Read(field_count, BufferCategory::kPlainBuffer);
70
71 if (field_count != RowType::size) {
72 throw CompositeSizeMismatch(field_count, RowType::size,
73 compiler::GetTypeName<T>());
74 }
75
76 ReadTuple(buffer, categories, RowType::GetTuple(this->value),
77 IndexSequence{});
78 }
79
80 private:
81 static constexpr std::size_t int_size = sizeof(Integer);
82
83 template <typename U>
84 void ReadField(FieldBuffer& buffer, const TypeBufferCategory& categories,
85 U& val) const {
86 Integer field_type = 0;
87 buffer.Read(field_type, BufferCategory::kPlainBuffer);
88 auto elem_category = GetTypeBufferCategory(categories, field_type);
89 if (elem_category == BufferCategory::kNoParser) {
90 throw LogicError{"Buffer category for oid " + std::to_string(field_type) +
91 " is unknown"};
92 }
93 buffer.ReadRaw(val, categories, elem_category);
94 }
95 template <typename Tuple, std::size_t... Indexes>
96 void ReadTuple(FieldBuffer& buffer, const TypeBufferCategory& categories,
97 Tuple&& tuple, std::index_sequence<Indexes...>) const {
98 (ReadField(buffer, categories,
99 std::get<Indexes>(std::forward<Tuple>(tuple))),
100 ...);
101 }
102};

104template <typename T>
105struct CompositeBinaryFormatter : BufferFormatterBase<T> {
106 using BaseType = BufferFormatterBase<T>;
107 using PgMapping = CppToPg<T>;
108 using RowType = io::RowType<T>;
109 using IndexSequence = typename RowType::IndexSequence;
110 static constexpr std::size_t size = RowType::size;
111
112 using BaseType::BaseType;
113
114 template <typename Buffer>
115 void operator()(const UserTypes& types, Buffer& buffer) const {
116 const auto& type_desc =
117 types.GetCompositeDescription(PgMapping::GetOid(types));
118 if (type_desc.Size() != size) {
119 throw CompositeSizeMismatch{type_desc.Size(), size,
120 compiler::GetTypeName<T>()};
121 }
122 // Number of fields
123 io::WriteBuffer(types, buffer, static_cast<Integer>(size));
124 WriteTuple(types, type_desc, buffer, RowType::GetTuple(this->value),
125 IndexSequence{});
126 }
127
128 private:
129 template <typename Buffer, typename U>
130 void WriteField(const UserTypes& types,
131 const CompositeTypeDescription& type_desc, std::size_t index,
132 Buffer& buffer, const U& val) const {
133 auto field_type = CppToPg<U>::GetOid(types);
134 const auto& field_desc = type_desc[index];
135 if (field_type != field_desc.type) {
136 if (io::MappedToSameType(static_cast<PredefinedOids>(field_type),
137 static_cast<PredefinedOids>(
138 types.FindDomainBaseOid(field_desc.type)))) {
139 field_type = field_desc.type;
140 } else {
141 throw CompositeMemberTypeMismatch(
142 PgMapping::postgres_name.schema, PgMapping::postgres_name.name,
143 field_desc.name, field_desc.type, field_type);
144 }
145 }
146 io::WriteBuffer(types, buffer, static_cast<Integer>(field_type));
147 io::WriteRawBinary(types, buffer, val, field_type);
148 }
149 template <typename Buffer, typename Tuple, std::size_t... Indexes>
150 void WriteTuple(const UserTypes& types,
151 const CompositeTypeDescription& type_desc, Buffer& buffer,
152 Tuple&& tuple, std::index_sequence<Indexes...>) const {
153 (WriteField(types, type_desc, Indexes, buffer,
154 std::get<Indexes>(std::forward<Tuple>(tuple))),
155 ...);
156 }
157};

188template <typename T>
189constexpr bool AssertHasCompositeParsers() {
190 static_assert(kIsRowType<T>);
191 return AssertTupleHasParsers<typename io::RowType<T>::TupleType>::value;
192}

194template <typename T>
