35/// @ingroup userver_components
36///
37/// @brief Template class for a simple gRPC client
38///
39/// The component is used as a storage of a gRPC client if you're OK with
40/// generated client and don't need to wrap it. The client can be fetched using
41/// `GetClient` method.
42///
43/// Example usage:
44///
45/// ```cpp
46/// int main(...)
47/// {
48/// ...
49/// component_list.Append<ugrpc::client::SimpleClientComponent<HelloClient>>("hello-client");
50/// ...
51/// }
52///
53/// MyComponent::MyComponent(const components::ComponentConfig& config,
54/// const components::ComponentContext& context)
55/// {
56/// HelloClient& client = context.FindComponent<
57/// ugrpc::client::SimpleClientComponent<HelloClient>>("hello-client").GetClient();
58/// ... use client ...
59/// }
60/// ```
61///
62/// ## Static config options:
63/// Name | Description | Default value
64/// ---- | ----------- | -------------
65/// endpoint | URL of the gRPC service | --
66/// client-name | name of the gRPC server we talk to, for diagnostics | <uses the component name>
67/// factory-component | ClientFactoryComponent name to use for client creation | --
68
69// clang-format on
70
71template <typename Client>
72class SimpleClientComponent : public impl::SimpleClientComponentAny {
73public:
74 /// Main component's constructor.
75 SimpleClientComponent(const components::ComponentConfig& config, const components::ComponentContext& context)
76 : SimpleClientComponentAny(config, context),
77 client_(FindFactory(config, context).MakeClient<Client>(MakeClientSettings(config, nullptr))) {}
78
79 /// To use a ClientQos config,
80 SimpleClientComponent(
81 const components::ComponentConfig& config,
82 const components::ComponentContext& context,
83 const dynamic_config::Key<ClientQos>& client_qos

85 : SimpleClientComponentAny(config, context),
86 client_(FindFactory(config, context).MakeClient<Client>(MakeClientSettings(config, &client_qos))) {}
87
88 /// @@brief Get gRPC service client
89 Client& GetClient() { return client_; }

91private:
92 Client client_;
93};
