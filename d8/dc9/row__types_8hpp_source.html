76template <typename T>
77constexpr bool DetectIsSuitableRowType() {
78 using type = std::remove_cv_t<T>;
79 return std::is_class_v<type> && !std::is_empty_v<type> &&
80 std::is_aggregate_v<type> && !std::is_polymorphic_v<type> &&
81 !std::is_union_v<type> && !postgres::detail::kIsInStdNamespace<type> &&
82 !postgres::detail::kIsInBoostNamespace<type>;
83}

110template <typename T>
111struct RowCategory

163template <typename T>
164struct RowTypeImpl<T, traits::RowCategoryType::kTuple> {
165 using ValueType = T;
166 using TupleType = T;
167 static constexpr std::size_t size = std::tuple_size<TupleType>::value;
168 using IndexSequence = std::make_index_sequence<size>;
169
170 static TupleType& GetTuple(ValueType& v) { return v; }
171 static const TupleType& GetTuple(const ValueType& v) { return v; }
172};
173
174template <typename T>
175struct RowTypeImpl<T, traits::RowCategoryType::kAggregate> {
176 using ValueType = T;
177 using TupleType =
178 decltype(boost::pfr::structure_tie(std::declval<ValueType&>()));
179 static constexpr std::size_t size = std::tuple_size<TupleType>::value;
180
181 using IndexSequence = std::make_index_sequence<size>;
182 static TupleType GetTuple(ValueType& v) {
183 return boost::pfr::structure_tie(v);
184 }
185 static auto GetTuple(const ValueType& value) {
186 return boost::pfr::structure_to_tuple(value);
187 }
188};
189
190template <typename T>
191struct RowTypeImpl<T, traits::RowCategoryType::kIntrusiveIntrospection> {
192 using ValueType = T;
193 using TupleType = decltype(std::declval<ValueType&>().Introspect());
194 static constexpr std::size_t size = std::tuple_size<TupleType>::value;
195 using IndexSequence = std::make_index_sequence<size>;
196 using ConstRefTuple = typename traits::AddTupleConstRef<TupleType>::type;
197
198 static TupleType GetTuple(ValueType& v) { return v.Introspect(); }
199 static auto GetTuple(const ValueType& v) {
200 // const_cast here is to relieve users from burden of writing
201 // const-overloaded functions or static template Introspect functions.
202 /// NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
203 return ConstRefTuple{const_cast<ValueType&>(v).Introspect()};
204 }
205};
