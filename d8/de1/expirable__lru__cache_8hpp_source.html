170template <typename Key, typename Value, typename Hash, typename Equal>
171Value ExpirableLruCache<Key, Value, Hash, Equal>::Get(
172 const Key& key, const UpdateValueFunc& update_func, ReadMode read_mode) {
173 auto now = utils::datetime::SteadyNow();
174 auto opt_old_value = GetOptional(key, update_func);
175 if (opt_old_value) {
176 return std::move(*opt_old_value);
177 }
178
179 auto mutex = mutex_set_.GetMutexForKey(key);
180 std::lock_guard lock(mutex);
181 // Test one more time - concurrent ExpirableLruCache::Get()
182 // might have put the value
183 auto old_value = lru_.Get(key);
184 if (old_value && !IsExpired(old_value->update_time, now)) {
185 return std::move(old_value->value);
186 }
187
188 auto value = update_func(key);
189 if (read_mode == ReadMode::kUseCache) {
190 lru_.Put(key, {value, now});
191 }
192 return value;
193}

195template <typename Key, typename Value, typename Hash, typename Equal>
196std::optional<Value> ExpirableLruCache<Key, Value, Hash, Equal>::GetOptional(
197 const Key& key, const UpdateValueFunc& update_func) {
198 auto now = utils::datetime::SteadyNow();
199 auto old_value = lru_.Get(key);
200
201 if (old_value) {
202 if (!IsExpired(old_value->update_time, now)) {
203 impl::CacheHit(stats_);
204
205 if (ShouldUpdate(old_value->update_time, now)) {
206 UpdateInBackground(key, update_func);
207 }
208
209 return std::move(old_value->value);
210 } else {
211 impl::CacheStale(stats_);
212 }
213 }
214 impl::CacheMiss(stats_);
215
216 return std::nullopt;
217}

310template <typename Key, typename Value, typename Hash, typename Equal>
311void ExpirableLruCache<Key, Value, Hash, Equal>::UpdateInBackground(
312 const Key& key, UpdateValueFunc update_func) {
313 stats_.total.background_updates++;
314 stats_.recent.GetCurrentCounter().background_updates++;
315
316 // cache will wait for all detached tasks in ~ExpirableLruCache()
317 engine::AsyncNoSpan([token = wait_token_storage_.GetToken(), this, key,
318 update_func = std::move(update_func)] {
319 auto mutex = mutex_set_.GetMutexForKey(key);
320 std::unique_lock lock(mutex, std::try_to_lock);
321 if (!lock) {
322 // someone is updating the key right now
323 return;
324 }
325
326 auto now = utils::datetime::SteadyNow();
327 auto value = update_func(key);
328 lru_.Put(key, {value, now});
329 }).Detach();
330}

332template <typename Key, typename Value, typename Hash, typename Equal>

334 std::chrono::steady_clock::time_point update_time,
335 std::chrono::steady_clock::time_point now) const {
336 auto max_lifetime = max_lifetime_.load();
337 return max_lifetime.count() != 0 && update_time + max_lifetime < now;

340template <typename Key, typename Value, typename Hash, typename Equal>
341bool ExpirableLruCache<Key, Value, Hash, Equal>::ShouldUpdate(
342 std::chrono::steady_clock::time_point update_time,
343 std::chrono::steady_clock::time_point now) const {
344 auto max_lifetime = max_lifetime_.load();
345 return (background_update_mode_.load() == BackgroundUpdateMode::kEnabled) &&
346 max_lifetime.count() != 0 && update_time + max_lifetime / 2 < now;
347}
