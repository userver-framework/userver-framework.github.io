85 void SendEvent(Args... args) const {
86 std::lock_guard lock(event_mutex_);
87 auto listeners = listeners_.Lock();
88
89 std::vector<engine::TaskWithResult<void>> tasks;
90 tasks.reserve(listeners->size());
91
92 for (const auto& [_, listener] : *listeners) {
93 tasks.push_back(utils::Async(
94 listener.task_name,
95 [&, &callback = listener.callback] { callback(args...); }));
96 }
97
98 std::size_t i = 0;
99 for (const auto& [_, listener] : *listeners) {
100 impl::WaitForTask(listener.name, tasks[i++]);
101 }
102 }

114 void RemoveListener(FunctionId id, UnsubscribingKind kind) noexcept final {
115 engine::TaskCancellationBlocker blocker;
116 auto listeners = listeners_.Lock();
117 const auto iter = listeners->find(id);
118
119 if (iter == listeners->end()) {
120 impl::ReportNotSubscribed(Name());
121 return;
122 }
123
124 if (kind == UnsubscribingKind::kAutomatic) {
125 impl::ReportUnsubscribingAutomatically(name_, iter->second.name);
126 }
127 listeners->erase(iter);
128 }

130 AsyncEventSubscriberScope DoAddListener(FunctionId id, std::string_view name,
131 Function&& func) final {
132 auto listeners = listeners_.Lock();
133 auto task_name = impl::MakeAsyncChannelName(name_, name);
134 const auto [iterator, success] = listeners->emplace(
135 id, Listener{std::string{name}, std::move(func), std::move(task_name)});
136 if (!success) impl::ReportAlreadySubscribed(Name(), name);
137 return AsyncEventSubscriberScope(*this, id);
138 }
