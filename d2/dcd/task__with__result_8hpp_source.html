29class USERVER_NODISCARD TaskWithResult : public Task {

36 /// @brief Constructor, for internal use only
37 /// @param task_processor task processor used for execution of this task
38 /// @param importance specifies whether this task can be auto-cancelled

42 TaskWithResult(

53
54 TaskWithResult(TaskWithResult&&) noexcept = default;
55 TaskWithResult& operator=(TaskWithResult&&) noexcept = default;
56
57 /// @brief Returns (or rethrows) the result of task invocation.
58 /// After return from this method the task is not valid.
59 /// @throws WaitInterruptedException when `current_task::IsCancelRequested()`
60 /// and no TaskCancellationBlockers are present.
61 /// @throws TaskCancelledException
62 /// if no result is available because the task was cancelled
63 T Get() noexcept(false) {
64 UASSERT(wrapped_call_ptr_);
65 EnsureValid();
66
67 Wait();
68 if (GetState() == State::kCancelled) {
69 throw TaskCancelledException(CancellationReason());
70 }
71 Invalidate();
72 return std::exchange(wrapped_call_ptr_, nullptr)->Retrieve();
73 }
74
75 using Task::TryGetContextAccessor;
76
77 private:
78 void EnsureValid() const {
79 UINVARIANT(IsValid(),
80 "TaskWithResult::Get was called on an invalid task. Note that "
81 "Get invalidates self, so it must be called at most once "
82 "per task");
83 }
84
85 void RethrowErrorResult() const final { (void)wrapped_call_ptr_->Get(); }
86
87 std::shared_ptr<utils::impl::WrappedCall<T>> wrapped_call_ptr_;
88};
