11def addLoggingLevel(levelName, levelNum, methodName=None):
12 """
13 Comprehensively adds a new logging level to the `logging` module and the
14 currently configured logging class.
15
16 `levelName` becomes an attribute of the `logging` module with the value
17 `levelNum`. `methodName` becomes a convenience method for both `logging`
18 itself and the class returned by `logging.getLoggerClass()` (usually just
19 `logging.Logger`). If `methodName` is not specified, `levelName.lower()` is
20 used.
21
22 To avoid accidental clobberings of existing attributes, this method will
23 raise an `AttributeError` if the level name is already an attribute of the
24 `logging` module or if the method name is already present
25
26 Example
27 -------
28 >>> addLoggingLevel('TRACE', logging.DEBUG - 5)
29 >>> logging.getLogger(__name__).setLevel("TRACE")
30 >>> logging.getLogger(__name__).trace('that worked')
31 >>> logging.trace('so did this')
32 >>> logging.TRACE
33 5
34
35 """
36 if not methodName:
37 methodName = levelName.lower()
38
39 if hasattr(logging, levelName):
40 raise AttributeError(
41 '{} already defined in logging module'.format(levelName),
42 )
43 if hasattr(logging, methodName):
44 raise AttributeError(
45 '{} already defined in logging module'.format(methodName),
46 )
47 if hasattr(logging.getLoggerClass(), methodName):
48 raise AttributeError(
49 '{} already defined in logger class'.format(methodName),
50 )
51
52 # This method was inspired by the answers to Stack Overflow post
53 # http://stackoverflow.com/q/2183233/2988730, especially
54 # http://stackoverflow.com/a/13638084/2988730
55 def logForLevel(self, message, *args, **kwargs):
56 if self.isEnabledFor(levelNum):
57 self._log(levelNum, message, args, **kwargs)
58
59 def logToRoot(message, *args, **kwargs):
60 logging.log(levelNum, message, *args, **kwargs)
61
62 logging.addLevelName(levelNum, levelName)
63 setattr(logging, levelName, levelNum)
64 setattr(logging.getLoggerClass(), methodName, logForLevel)
65 setattr(logging, methodName, logToRoot)
66
