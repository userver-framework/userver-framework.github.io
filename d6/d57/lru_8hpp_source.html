181template <typename T, typename U, typename Hash, typename Eq>
182bool LruBase<T, U, Hash, Eq>::Put(const T& key, U value) {
183 auto it = map_.find(key, map_.hash_function(), map_.key_eq());
184 if (it != map_.end()) {
185 it->SetValue(std::move(value));
186 MarkRecentlyUsed(*it);
187 return false;
188 }
189
190 Add(key, std::move(value));
191 return true;
192}

229template <typename T, typename U, typename Hash, typename Eq>
230void LruBase<T, U, Hash, Eq>::SetMaxSize(size_t new_max_size) {
231 UASSERT(new_max_size > 0);
232 if (!new_max_size) ++new_max_size;
233
234 if (buckets_.size() == new_max_size) {
235 return;
236 }
237
238 while (map_.size() > new_max_size) {
239 ExtractNode(list_.begin());
240 }
241
242 std::vector<BucketType> new_buckets(new_max_size);
243 map_.rehash(BucketTraits(new_buckets.data(), new_max_size));
244 buckets_.swap(new_buckets);
245}

275template <typename T, typename U, typename Hash, typename Eq>
276U& LruBase<T, U, Hash, Eq>::Add(const T& key, U value) {
277 if (map_.size() < buckets_.size()) {
278 auto node = std::make_unique<Node>(T{key}, std::move(value));
279 return InsertNode(std::move(node)).GetValue();
280 }
281
282 auto node = ExtractNode(list_.begin());
283 node->SetKey(key);
284 node->SetValue(std::move(value));
285 return InsertNode(std::move(node)).GetValue();
286}

293template <typename T, typename U, typename Hash, typename Eq>
294std::unique_ptr<LruNode<T, U>> LruBase<T, U, Hash, Eq>::ExtractNode(
295 typename List::iterator it) noexcept {
296 UASSERT(it != list_.end());
297
298 std::unique_ptr<Node> ret(&*it);
299 map_.erase(map_.iterator_to(*it));
300 list_.erase(it);
301 return ret;
302}

304template <typename T, typename U, typename Hash, typename Eq>
305LruNode<T, U>& LruBase<T, U, Hash, Eq>::InsertNode(
306 std::unique_ptr<impl::LruNode<T, U>>&& node) noexcept {
307 UASSERT(node);
308
309 auto [it, ok] = map_.insert(*node); // noexcept
310 UASSERT(ok);
311 list_.insert(list_.end(), *node); // noexcept
312
313 return *node.release();
