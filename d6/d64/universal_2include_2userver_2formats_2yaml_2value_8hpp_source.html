194 /// For example:

317auto Value::As(First&& default_arg, Rest&&... more_default_args) const {
318 if (IsMissing() || IsNull()) {
319 // intended raw ctor call, sometimes casts
320 // NOLINTNEXTLINE(google-readability-casting)
321 return decltype(As<T>())(std::forward<First>(default_arg), std::forward<Rest>(more_default_args)...);
322 }
323 return As<T>();
324}
