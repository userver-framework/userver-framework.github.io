36/// @brief Binary formatter for utils::datetime::Date
37template <>
38struct BufferFormatter<Date> {
39 const Date value;
40
41 explicit BufferFormatter(Date value) : value{value} {}
42
43 template <typename Buffer>
44 void operator()(const UserTypes& types, Buffer& buffer) {
45 static const auto kPgEpoch = PostgresEpochDate();
46 if (value == kDatePositiveInfinity) {
47 WriteBuffer(types, buffer, std::numeric_limits<Integer>::max());
48 } else if (value == kDateNegativeInfinity) {
49 WriteBuffer(types, buffer, std::numeric_limits<Integer>::min());
50 } else {
51 auto pg_days = static_cast<Integer>(
52 (value.GetSysDays() - kPgEpoch.GetSysDays()).count());
53 WriteBuffer(types, buffer, pg_days);
54 }
55 }
56};

67 Integer pg_days{0};
68 ReadBuffer(buffer, pg_days);
69 if (pg_days == std::numeric_limits<Integer>::max()) {
