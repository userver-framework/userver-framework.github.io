26/// Helper template. Default implementation is straightforward.
27template <typename T>
28struct QueueHelper {
29 using LockFreeQueue = boost::lockfree::queue<T>;
30
31 static void Push(LockFreeQueue& queue, T&& value) {
32 [[maybe_unused]] bool push_result = queue.push(std::move(value));
33 UASSERT(push_result);
34 }
35
36 [[nodiscard]] static bool Pop(LockFreeQueue& queue, T& value) { return queue.pop(value); }
37
38 static_assert(
39 std::is_trivially_destructible_v<T>,
40 "T has non-trivial destructor. Use "
41 "MpscQueue<std::unique_ptr<T>> instead of MpscQueue<T>"
42 );
43};
44
45/// This partial specialization allows to use std::unique_ptr with Queue.
46template <typename T>
47struct QueueHelper<std::unique_ptr<T>> {
48 using LockFreeQueue = boost::lockfree::queue<T*>;
49
50 static void Push(LockFreeQueue& queue, std::unique_ptr<T>&& value) {
51 QueueHelper<T*>::Push(queue, value.release());
52 }
53
54 [[nodiscard]] static bool Pop(LockFreeQueue& queue, std::unique_ptr<T>& value) {
55 T* ptr{nullptr};
56 if (!QueueHelper<T*>::Pop(queue, ptr)) return false;
57 value.reset(ptr);
58 return true;
59 }
60};

114 /// Get a `Producer` which makes it possible to push items into the queue.
115 /// Can be called multiple times. The resulting `Producer` is not thread-safe,
116 /// so you have to use multiple Producers of the same queue to simultaneously
117 /// write from multiple coroutines/threads.
118 ///
119 /// @note `Producer` may outlive the queue and the `Consumer`.
120 Producer GetProducer();
121
122 /// Get a `MultiProducer` which makes it possible to push items into the
123 /// queue. Can be called multiple times. The resulting `MultiProducer` is
124 /// thread-safe, so it can be used simultaneously from multiple
125 /// coroutines/threads.
126 ///
127 /// @note `MultiProducer` may outlive the queue and the `Consumer`.
128 MultiProducer GetMultiProducer();
129
130 /// Get a `Consumer` which makes it possible to read items from the queue.
131 /// Can be called only once. You may not use the `Consumer` simultaneously
132 /// from multiple coroutines/threads.
133 ///
134 /// @note `Consumer` may outlive the queue and producers.
135 Consumer GetConsumer();

160 // specialization. In that case, resolves to boost::lockfree::queue<T*>
161 typename QueueHelper::LockFreeQueue queue_{1};
162 engine::SingleConsumerEvent nonempty_event_;

172template <typename T>
173MpscQueue<T>::~MpscQueue() {
174 UASSERT(consumer_is_created_and_dead_ || !consumer_is_created_);
175 UASSERT(!producers_count_);
176 // Clear remaining items in queue. This will work for unique_ptr as well.
177 T value;
178 ConsumerToken temp_token{queue_};
179 while (PopNoblock(temp_token, value)) {
180 }
181}

184typename MpscQueue<T>::Producer MpscQueue<T>::GetProducer() {

186 producer_is_created_and_dead_ = false;

223}

225template <typename T>
226bool MpscQueue<T>::PushNoblock(ProducerToken& token, T&& value) {

230template <typename T>
231bool MpscQueue<T>::DoPush(ProducerToken& /*unused*/, T&& value) {
232 if (consumer_is_created_and_dead_) {
233 remaining_capacity_.unlock_shared();
234 return false;
235 }
236
237 QueueHelper::Push(queue_, std::move(value));
238 ++size_;
239 nonempty_event_.Send();
240
241 return true;
242}
243
244template <typename T>
245bool MpscQueue<T>::Pop(ConsumerToken& token, T& value, engine::Deadline deadline) {
246 while (!DoPop(token, value)) {
247 if (producer_is_created_and_dead_ || !nonempty_event_.WaitForEventUntil(deadline)) {
248 // Producer might have pushed something in queue between .pop()
249 // and !producer_is_created_and_dead_ check. Check twice to avoid
250 // TOCTOU.
251 return DoPop(token, value);
252 }
253 }
254 return true;
255}
256
257template <typename T>
258bool MpscQueue<T>::PopNoblock(ConsumerToken& token, T& value) {
259 return DoPop(token, value);
260}
261
262template <typename T>
263bool MpscQueue<T>::DoPop(ConsumerToken& /*unused*/, T& value) {
264 if (QueueHelper::Pop(queue_, value)) {
265 --size_;
266 remaining_capacity_.unlock_shared();
267 nonempty_event_.Reset();
268 return true;
269 }
270 return false;
271}

273template <typename T>

279template <typename T>
280void MpscQueue<T>::MarkProducerIsDead() {

282 nonempty_event_.Send();
