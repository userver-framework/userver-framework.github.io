380template <typename T, typename First, typename... Rest>
381auto Value::As(First&& default_arg, Rest&&... more_default_args) const {
382 if (IsMissing() || IsNull()) {
383 // intended raw ctor call, sometimes casts
384 // NOLINTNEXTLINE(google-readability-casting)
385 return decltype(As<T>())(std::forward<First>(default_arg), std::forward<Rest>(more_default_args)...);
386 }
387 return As<T>();
388}

390template <typename T>
391auto Value::As(Value::DefaultConstructed) const {
392 return (IsMissing() || IsNull()) ? decltype(As<T>())() : As<T>();

395template <typename T>
396T Value::ConvertTo() const {
397 if constexpr (formats::common::impl::kHasConvert<Value, T>) {
398 return Convert(*this, formats::parse::To<T>{});
399 } else if constexpr (formats::common::impl::kHasParse<Value, T>) {
400 return Parse(*this, formats::parse::To<T>{});
401 } else {
402 static_assert(
403 !sizeof(T),
404 "There is no `Convert(const Value&, formats::parse::To<T>)` or"
405 "`Parse(const Value&, formats::parse::To<T>)`"
406 "in namespace of `T` or `formats::parse`. "
407 "Probably you have not provided a `Convert` function overload."
408 );
409 }
410}

413T Value::ConvertTo(First&& default_arg, Rest&&... more_default_args) const {
