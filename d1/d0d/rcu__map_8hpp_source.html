33/// Default RcuMap traits.
34/// Member types:
35/// - `Hash` is a functor type that returns hash value for `Key`
36/// - `keyEqual` is a functor type that provide equality test for two values of
37/// type `Key`
38/// - `MutexType` is a writer's mutex type that has to be used to protect
39/// structure on update
40template <typename Key, typename Value>
41struct DefaultRcuMapTraits {
42 using Hash = std::hash<Key>;
43 using KeyEqual = std::equal_to<Key>;
44 using MutexType = engine::Mutex;
45};

88/// @ingroup userver_concurrency userver_containers
89///
90/// @brief Map-like structure allowing RCU keyset updates.
91///
92/// Only keyset changes are thread-safe in scope of this class.
93/// Values are stored in `shared_ptr`s and are not copied during keyset change.
94/// The map itself is implemented as rcu::Variable, so every keyset change
95/// (e.g. insert or erase) triggers the whole map copying.
96/// @note No synchronization is provided for value access, it must be
97/// implemented by Value when necessary.
98///
99/// ## Example usage:
100///
101/// @snippet rcu/rcu_map_test.cpp Sample rcu::RcuMap usage
102///
103/// @see @ref scripts/docs/en/userver/synchronization.md

106 using RcuTraits = typename impl::RcuTraitsFromRcuMapTraits<RcuMapTraits>;
107
108public:
109 static_assert(!std::is_reference_v<Key>);
110 static_assert(!std::is_reference_v<Value>);
111 static_assert(!std::is_const_v<Key>);

287template <typename K, typename V, typename RcuMapTraits>
288// Protects from assignment to map[key]
289// NOLINTNEXTLINE(readability-const-return-type)
290const typename RcuMap<K, V, RcuMapTraits>::ValuePtr RcuMap<K, V, RcuMapTraits>::operator[](const K& key) {
291 auto value = Get(key);
292 if (!value) {
293 auto txn = rcu_.StartWrite();
294 auto insertion_result = txn->emplace(key, std::make_shared<V>());
295 value = insertion_result.first->second;
296 if (insertion_result.second) txn.Commit();
297 }
298 return value;
299}

320template <typename K, typename V, typename RcuMapTraits>
321typename RcuMap<K, V, RcuMapTraits>::InsertReturnType
322RcuMap<K, V, RcuMapTraits>::DoInsert(const K& key, typename RcuMap<K, V, RcuMapTraits>::ValuePtr value) {
323 auto txn = rcu_.StartWrite();
324 auto insertion_result = txn->emplace(key, std::move(value));
325 InsertReturnType result{insertion_result.first->second, insertion_result.second};
326 if (result.inserted) txn.Commit();
327 return result;
328}

330template <typename K, typename V, typename RcuMapTraits>
331template <typename... Args>
332typename RcuMap<K, V, RcuMapTraits>::InsertReturnType
333RcuMap<K, V, RcuMapTraits>::TryEmplace(const K& key, Args&&... args) {
334 InsertReturnType result{Get(key), false};
335 if (!result.value) {
336 auto txn = rcu_.StartWrite();
337 auto insertion_result = txn->try_emplace(key, nullptr);
338 if (insertion_result.second) {
339 result.value = insertion_result.first->second = std::make_shared<V>(std::forward<Args>(args)...);
340 txn.Commit();
341 result.inserted = true;
342 } else {
343 result.value = insertion_result.first->second;
344 }
345 }
346 return result;
347}
