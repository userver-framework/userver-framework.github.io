262template <typename K, typename V>
263// Protects from assignment to map[key]
264// NOLINTNEXTLINE(readability-const-return-type)
265const typename RcuMap<K, V>::ValuePtr RcuMap<K, V>::operator[](const K& key) {
266 auto value = Get(key);
267 if (!value) {
268 auto txn = rcu_.StartWrite();
269 auto insertion_result = txn->emplace(key, std::make_shared<V>());
270 value = insertion_result.first->second;
271 if (insertion_result.second) txn.Commit();
272 }
273 return value;

295template <typename K, typename V>
296typename RcuMap<K, V>::InsertReturnType RcuMap<K, V>::DoInsert(
297 const K& key, typename RcuMap<K, V>::ValuePtr value) {
298 auto txn = rcu_.StartWrite();
299 auto insertion_result = txn->emplace(key, std::move(value));
300 InsertReturnType result{insertion_result.first->second,
301 insertion_result.second};
302 if (result.inserted) txn.Commit();
303 return result;
304}

306template <typename K, typename V>
307template <typename... Args>
308typename RcuMap<K, V>::InsertReturnType RcuMap<K, V>::TryEmplace(
309 const K& key, Args&&... args) {
310 InsertReturnType result{Get(key), false};
311 if (!result.value) {
312 auto txn = rcu_.StartWrite();
313 auto insertion_result = txn->try_emplace(key, nullptr);
314 if (insertion_result.second) {
315 result.value = insertion_result.first->second =
316 std::make_shared<V>(std::forward<Args>(args)...);
317 txn.Commit();
318 result.inserted = true;
319 } else {
320 result.value = insertion_result.first->second;
321 }
322 }
323 return result;
324}
