578 def value_at(
579 self,
580 subpath: typing.Optional[str] = None,
581 add_labels: typing.Optional[typing.Dict] = None,
582 *,
583 default: typing.Optional[float] = None,
584 ) -> metric_module.MetricValue:
585 """
586 Returns a single metric value at the specified path, prepending
587 the path provided at construction. If a dict of labels is provided,
588 does en exact match of labels, prepending the labels provided
589 at construction.
590
591 @param subpath Suffix of the metric path; the path provided
592 at construction is prepended
593 @param add_labels Labels that the metric must have in addition
594 to the labels provided at construction
595 @param default An optional default value in case the metric is missing
596 @throws AssertionError if not one metric by path
597 """
598 base_path = self._path or self._prefix
599 if base_path and subpath:
600 path = f'{base_path}.{subpath}'
601 else:
602 assert base_path or subpath, 'No path provided'
603 path = base_path or subpath or ''
604 labels: typing.Optional[dict] = None
605 if self._labels is not None or add_labels is not None:
606 labels = {**(self._labels or {}), **(add_labels or {})}
607 return self.diff.value_at(path, labels, default=default)

629def _subtract_metrics_snapshots(
630 current: metric_module.MetricsSnapshot,
631 initial: metric_module.MetricsSnapshot,
632 diff_gauge: bool,
633) -> metric_module.MetricsSnapshot:
634 return metric_module.MetricsSnapshot({
635 path: {
636 _subtract_metrics(path, current_metric, initial, diff_gauge)
637 for current_metric in current_group
638 }
639 for path, current_group in current.items()
640 })

643def _subtract_metrics(
644 path: str,
645 current_metric: metric_module.Metric,
646 initial: metric_module.MetricsSnapshot,
647 diff_gauge: bool,
648) -> metric_module.Metric:
649 initial_group = initial.get(path, None)
650 if initial_group is None:
651 return current_metric
652 initial_metric = next(
653 (x for x in initial_group if x.labels == current_metric.labels), None,
654 )
655 if initial_metric is None:
656 return current_metric
657
658 return metric_module.Metric(
659 labels=current_metric.labels,
660 value=_subtract_metric_values(
661 current=current_metric,
662 initial=initial_metric,
663 diff_gauge=diff_gauge,
664 ),
665 _type=current_metric.type(),
666 )

669def _subtract_metric_values(
670 current: metric_module.Metric,
671 initial: metric_module.Metric,
672 diff_gauge: bool,
673) -> metric_module.MetricValue:
674 assert current.type() is not metric_module.MetricType.UNSPECIFIED
675 assert initial.type() is not metric_module.MetricType.UNSPECIFIED
676 assert current.type() == initial.type()
677
678 if isinstance(current.value, metric_module.Histogram):
679 assert isinstance(initial.value, metric_module.Histogram)
680 return _subtract_metric_values_hist(current=current, initial=initial)
681 else:
682 assert not isinstance(initial.value, metric_module.Histogram)
683 return _subtract_metric_values_num(
684 current=current, initial=initial, diff_gauge=diff_gauge,
685 )

687
688def _subtract_metric_values_num(
689 current: metric_module.Metric,
690 initial: metric_module.Metric,
691 diff_gauge: bool,
692) -> float:
693 current_value = typing.cast(float, current.value)
694 initial_value = typing.cast(float, initial.value)
695 should_diff = (
696 current.type() is metric_module.MetricType.RATE
697 or initial.type() is metric_module.MetricType.RATE
698 or diff_gauge
699 )
700 return current_value - initial_value if should_diff else current_value
701
702
703def _subtract_metric_values_hist(
704 current: metric_module.Metric, initial: metric_module.Metric,
705) -> metric_module.Histogram:
706 current_value = typing.cast(metric_module.Histogram, current.value)
707 initial_value = typing.cast(metric_module.Histogram, initial.value)
708 assert current_value.bounds == initial_value.bounds
709 return metric_module.Histogram(
710 bounds=current_value.bounds,
711 buckets=[
712 t[0] - t[1]
713 for t in zip(current_value.buckets, initial_value.buckets)
714 ],
715 inf=current_value.inf - initial_value.inf,
716 )

1224 """
1225 Used for computing the requests that we need to automatically align
1226 the service state with the test fixtures state.
1227 """
1228
1229 def __init__(
1230 self,
1231 *,
1232 mocked_time,
1233 testpoint,
1234 testpoint_control,
1235 invalidation_state: caches.InvalidationState,
1236 cache_control: typing.Optional[caches.CacheControl],
1237 ):
1238 self._state = _State(
1239 invalidation_state=copy.deepcopy(invalidation_state),
1240 )
1241 self._mocked_time = mocked_time

1247 @contextlib.contextmanager
1248 def updating_state(self, body: typing.Dict[str, typing.Any]):
1249 """
1250 Whenever `tests_control` handler is invoked
1251 (by the client itself during `prepare` or manually by the user),
1252 we need to synchronize `_state` with the (supposed) service state.
1253 The state update is decoded from the request body.
1254 """
1255 saved_state = copy.deepcopy(self._state)
1256 try:
1257 self._update_state(body)
1258 self._apply_new_state()
1259 yield
1260 except Exception: # noqa
1261 self._state = saved_state
1262 self._apply_new_state()
1263 raise
1264
1265 def get_pending_update(self) -> typing.Dict[str, typing.Any]:
1266 """
1267 Compose the body of the `tests_control` request required to completely
1268 synchronize the service state with the state of test fixtures.
1269 """
1270 body: typing.Dict[str, typing.Any] = {}
1271
1272 if self._invalidation_state.has_caches_to_update:
1273 body['invalidate_caches'] = {'update_type': 'full'}
1274 if not self._invalidation_state.should_update_all_caches:
1275 body['invalidate_caches']['names'] = list(
1276 self._invalidation_state.caches_to_update,
1277 )
1278
1279 desired_testpoints = self._testpoint.keys()
1280 if self._state.testpoints != frozenset(desired_testpoints):
1281 body['testpoints'] = sorted(desired_testpoints)
1282
1283 desired_now = self._get_desired_now()
1284 if self._state.now != desired_now:
1285 body['mock_now'] = desired_now
1286
1287 return body

1289 @contextlib.contextmanager
1290 def cache_control_update(self) -> typing.ContextManager[typing.Dict]:
1291 pending_update = self.get_pending_update()
1292 invalidate_caches = pending_update.get('invalidate_caches')
1293 if not invalidate_caches or not self._cache_control:
1294 yield pending_update
1295 else:
1296 cache_names = invalidate_caches.get('names')
1297 staged, actions = self._cache_control.query_caches(cache_names)
1298 self._apply_cache_control_actions(invalidate_caches, actions)
1299 yield pending_update
1300 self._cache_control.commit_staged(staged)

1302 @staticmethod
1303 def _apply_cache_control_actions(
1304 invalidate_caches: typing.Dict,
1305 actions: typing.List[typing.Tuple[str, caches.CacheControlAction]],
1306 ) -> None:
1307 cache_names = invalidate_caches.get('names')
1308 exclude_names = invalidate_caches.setdefault('exclude_names', [])
1309 force_incremental_names = invalidate_caches.setdefault(
1310 'force_incremental_names', [],
1311 )
1312 for cache_name, action in actions:
1313 if action == caches.CacheControlAction.INCREMENTAL:
1314 force_incremental_names.append(cache_name)
1315 elif action == caches.CacheControlAction.EXCLUDE:
1316 if cache_names is not None:
1317 cache_names.remove(cache_name)
1318 else:
1319 exclude_names.append(cache_name)

1321 def _update_state(self, body: typing.Dict[str, typing.Any]) -> None:
1322 body_invalidate_caches = body.get('invalidate_caches', {})
1323 update_type = body_invalidate_caches.get('update_type', 'full')
1324 body_cache_names = body_invalidate_caches.get('names', None)
1325 # An incremental update is considered insufficient to bring a cache
1326 # to a known state.
1327 if body_invalidate_caches and update_type == 'full':
1328 if body_cache_names is None:
1329 self._state.invalidation_state.on_all_caches_updated()
1330 else:
1331 self._state.invalidation_state.on_caches_updated(
1332 body_cache_names,
1333 )
1334
1335 if 'mock_now' in body:
1336 self._state.now = body['mock_now']
1337
1338 testpoints: typing.Optional[typing.List[str]] = body.get(
1339 'testpoints', None,
1340 )
1341 if testpoints is not None:
1342 self._state.testpoints = frozenset(testpoints)

1344 def _apply_new_state(self):
1345 """Apply new state to related components."""
1346 self._testpoint_control.enabled_testpoints = self._state.testpoints
1347 self._invalidation_state.assign_copy(self._state.invalidation_state)

1350 if self._mocked_time.is_enabled:
