546@dataclasses.dataclass(frozen=True)
547class State:
548 caches_invalidated: bool = False
549 now: typing.Optional[str] = _UNKNOWN_STATE
550 testpoints: typing.FrozenSet[str] = frozenset([_UNKNOWN_STATE])

553class StateManager:
554 def __init__(
555 self,
556 *,
557 mocked_time,
558 testpoint,
559 testpoint_control,
560 cache_blocklist: typing.List[str],
561 ):
562 self._state = State()
563 self._mocked_time = mocked_time
564 self._testpoint = testpoint
565 self._testpoint_control = testpoint_control
566 self._cache_blocklist = cache_blocklist

568 @contextlib.contextmanager
569 def updating_state(self, body):
570 saved_state = self._state
571 try:
572 self._state = self._update_state(body)
573 self._apply_new_state()
574 yield
575 except: # noqa
576 self._state = saved_state
577 self._apply_new_state()
578 raise

580 def get_pending_update(self) -> typing.Dict[str, typing.Any]:
581 """Get arguments to be passed in ``POST testpoint`` to sync service
582 state with desired state
583 """
584 state = self._get_desired_state()
585 body: typing.Dict[str, typing.Any] = {}
586
587 if self._state.caches_invalidated != state.caches_invalidated:
588 body['invalidate_caches'] = {
589 'update_type': 'full',
590 'names_blocklist': self._cache_blocklist,
591 }
592
593 if self._state.testpoints != state.testpoints:
594 body['testpoints'] = sorted(state.testpoints)
595
596 if self._state.now != state.now:
597 body['mock_now'] = state.now
598
599 return body
600
601 def _update_state(self, body: dict) -> State:
602 update: typing.Dict[str, typing.Any] = {}
603
604 invalidate_caches = body.get('invalidate_caches', {})
605 update_type = invalidate_caches.get('update_type', 'full')
606 names = invalidate_caches.get('names', None)
607 if invalidate_caches and update_type == 'full' and not names:
608 update['caches_invalidated'] = True
609
610 if 'mock_now' in body:
611 update['now'] = body['mock_now']
612
613 testpoints = body.get('testpoints', None)
614 if testpoints is not None:
615 update['testpoints'] = frozenset(testpoints)
616
617 return dataclasses.replace(self._state, **update)
618
619 def _apply_new_state(self):
620 """Apply new state to related components."""
621 self._testpoint_control.enabled_testpoints = self._state.testpoints
622
623 def _get_desired_state(self) -> State:
624 if self._mocked_time.is_enabled:
625 now = utils.timestring(self._mocked_time.now())
626 else:
627 now = None
628 return State(
629 caches_invalidated=True,
630 testpoints=frozenset(self._testpoint.keys()),
631 now=now,
632 )

635async def _task_check_response(name: str, response) -> dict:
636 async with response:
637 if response.status == 404:
638 raise TestsuiteTaskNotFound(f'Testsuite task {name!r} not found')
639 if response.status == 409:
640 raise TestsuiteTaskConflict(f'Testsuite task {name!r} conflict')
641 assert response.status == 200
642 data = await response.json()
643 if not data.get('status', True):
644 raise TestsuiteTaskFailed(name, data['reason'])
645 return data
