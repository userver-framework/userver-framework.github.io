158template <typename T>
159CachingComponentBase<T>::~CachingComponentBase() {
160 // Avoid a deadlock in WaitForAllTokens
161 cache_.Assign(nullptr);
162 // We must wait for destruction of all instances of T to finish, otherwise
163 // it's UB if T's destructor accesses dependent components
164 wait_token_storage_.WaitForAllTokens();

198template <typename T>
199void CachingComponentBase<T>::Set(std::unique_ptr<const T> value_ptr) {
200 auto deleter = [token = wait_token_storage_.GetToken(),
201 &cache_task_processor =
202 GetCacheTaskProcessor()](const T* raw_ptr) mutable {
203 std::unique_ptr<const T> ptr{raw_ptr};
204
205 // Kill garbage asynchronously as T::~T() might be very slow
206 engine::CriticalAsyncNoSpan(cache_task_processor, [ptr = std::move(ptr),
207 token = std::move(
208 token)]() mutable {
209 // Make sure *ptr is deleted before token is destroyed
210 ptr.reset();
211 }).Detach();
212 };
213
214 const std::shared_ptr<const T> new_value(value_ptr.release(),
215 std::move(deleter));
216 cache_.Assign(new_value);
217 event_channel_.SendEvent(new_value);
218 OnCacheModified();
219}
