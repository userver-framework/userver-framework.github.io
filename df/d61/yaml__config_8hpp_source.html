150template <typename T, typename First, typename... Rest>
151T YamlConfig::As(First&& default_arg, Rest&&... more_default_args) const {
152 if (IsMissing()) {
153 // intended raw ctor call, sometimes casts
154 // NOLINTNEXTLINE(google-readability-casting)
155 return T(std::forward<First>(default_arg),
156 std::forward<Rest>(more_default_args)...);
157 }
158 return As<T>();
159}
