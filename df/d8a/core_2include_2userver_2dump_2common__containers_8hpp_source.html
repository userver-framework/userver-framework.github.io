82template <typename VariantType>
83VariantType ReadVariant(Reader& reader, std::size_t index) {
84 static constexpr auto VariantSize = std::variant_size_v<VariantType>;
85 std::optional<VariantType> result;
86
87 utils::WithConstexprIndex<VariantSize>(index, [&](auto index_constant) {
88 static constexpr auto kIndex = decltype(index_constant)::value;
89 using Alternative = std::variant_alternative_t<kIndex, VariantType>;
90 // Not using ReadLazyPrvalue because of stdlib issues on some compilers.
91 result.emplace(std::in_place_index<kIndex>, reader.Read<Alternative>());
92 });
93
94 return std::move(*result);
95}

99/// @brief Container serialization support
100template <typename T>
101std::enable_if_t<kIsContainer<T> && kIsWritable<meta::RangeValueType<T>>> Write(Writer& writer, const T& value) {
102 writer.Write(std::size(value));
103 for (const auto& item : value) {
104 // explicit cast for vector<bool> shenanigans
105 writer.Write(static_cast<const meta::RangeValueType<T>&>(item));
106 }
107}

109/// @brief Container deserialization support
110template <typename T>
111std::enable_if_t<kIsContainer<T> && kIsReadable<meta::RangeValueType<T>>, T> Read(Reader& reader, To<T>) {
112 const auto size = reader.Read<std::size_t>();
113 T result{};
114 if constexpr (meta::kIsReservable<T>) {
115 result.reserve(size);
116 }
117 for (std::size_t i = 0; i < size; ++i) {
118 dump::Insert(result, reader.Read<meta::RangeValueType<T>>());
119 }
120 return result;
121}

220/// @brief `boost::bimap` serialization support
221template <typename L, typename R, typename... Args>
222std::enable_if_t<
223 kIsWritable<impl::BoostBimapLeftKey<L, R, Args...>> && kIsWritable<impl::BoostBimapRightKey<L, R, Args...>>>
224Write(Writer& writer, const boost::bimap<L, R, Args...>& map) {
225 writer.Write(map.size());
226
227 for (const auto& [left, right] : map) {
228 writer.Write(left);
229 writer.Write(right);
230 }
231}

233/// @brief `boost::bimap` deserialization support
234template <typename L, typename R, typename... Args>
235std::enable_if_t<
236 kIsReadable<impl::BoostBimapLeftKey<L, R, Args...>> && kIsReadable<impl::BoostBimapRightKey<L, R, Args...>>,
237 boost::bimap<L, R, Args...>>
238Read(Reader& reader, To<boost::bimap<L, R, Args...>>) {
239 using BoostBimap = impl::BoostBimap<L, R, Args...>;
240
241 using BoostBimapLeftKey = impl::BoostBimapLeftKey<L, R, Args...>;
242 using BoostBimapRightKey = impl::BoostBimapRightKey<L, R, Args...>;
243
244 using BoostBimapSizeType = typename BoostBimap::size_type;
245
246 BoostBimap map;
247 // bimap doesn't have reserve :(
248
249 const auto size = reader.Read<BoostBimapSizeType>();
250 for (BoostBimapSizeType i = 0; i < size; ++i) {
251 // `Read`s are guaranteed to occur left-to-right in brace-init
252 map.insert({
253 reader.Read<BoostBimapLeftKey>(),
254 reader.Read<BoostBimapRightKey>(),
255 });
256 }
257
258 return map;
259}

270/// @brief `boost::multi_index_container` deserialization support
271template <typename T, typename Index, typename Alloc>
272std::enable_if_t<kIsReadable<T>, boost::multi_index_container<T, Index, Alloc>>
273Read(Reader& reader, To<boost::multi_index_container<T, Index, Alloc>>) {
274 const auto size = reader.Read<std::size_t>();
275 boost::multi_index_container<T, Index, Alloc> container;
276
277 // boost::multi_index_container has reserve() with some, but not all, configs
278 if constexpr (meta::kIsReservable<boost::multi_index_container<T, Index, Alloc>>) {
279 container.reserve(size);
280 }
281
282 for (std::size_t i = 0; i < size; ++i) {
283 container.insert(reader.Read<T>());
284 }
285
286 return container;
287}
