79template <typename VariantType>
80VariantType ReadVariant(Reader& reader, std::size_t index) {
81 std::optional<VariantType> result;
82 using Indices = std::make_index_sequence<std::variant_size_v<VariantType>>;
83
84 ForEachIndex(Indices{}, [&](auto index_constant) {
85 static constexpr auto kIndex = decltype(index_constant)::value;
86 using Alternative = std::variant_alternative_t<kIndex, VariantType>;
87 if (index == kIndex) {
88 // Not using ReadLazyPrvalue because of stdlib issues on some compilers.
89 result.emplace(std::in_place_index<kIndex>, reader.Read<Alternative>());
90 }
91 });
92
93 return std::move(*result);
94}

276Read(Reader& reader, To<boost::multi_index_container<T, Index, Alloc>>) {
277 const auto size = reader.Read<std::size_t>();
