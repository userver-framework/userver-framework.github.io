107template <typename... Args>
108CmdArgs& CmdArgs::Then(Args&&... _args) {
109 args.emplace_back();
110 auto& new_args = args.back();
111 new_args.reserve(sizeof...(Args));
112 (PutArg(new_args, std::forward<Args>(_args)), ...);
113 return *this;
