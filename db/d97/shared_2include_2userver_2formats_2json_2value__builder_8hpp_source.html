184template <typename T>
185Value ValueBuilder::DoSerialize(const T& t) {
186 static_assert(
187 formats::common::impl::kHasSerialize<Value, T>,
188 "There is no `Serialize(const T&, formats::serialize::To<json::Value>)` "
189 "in namespace of `T` or `formats::serizalize`. "
190 ""
191 "Probably you forgot to include the "
192 "<userver/formats/serialize/common_containers.hpp> header "
193 "or one of the <formats/json/serialize_*.hpp> headers or you "
194 "have not provided a `Serialize` function overload.");
195
196 return Serialize(t, formats::serialize::To<Value>());
197}

199template <typename T>
200std::enable_if_t<std::is_integral<T>::value && sizeof(T) <= sizeof(int64_t),
201 Value>
202Serialize(T value, formats::serialize::To<Value>) {
203 using Type = std::conditional_t<std::is_signed<T>::value, int64_t, uint64_t>;
204 return json::ValueBuilder(static_cast<Type>(value)).ExtractValue();
205}
