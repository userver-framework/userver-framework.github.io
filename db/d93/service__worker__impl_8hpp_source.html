80template <typename GrpcppService, typename CallTraits>
81class CallData final {
82 public:
83 explicit CallData(const MethodData<GrpcppService, CallTraits>& method_data)
84 : wait_token_(method_data.service_data.wait_tokens.GetToken()),
85 method_data_(method_data) {
86 UASSERT(method_data.method_id <
87 method_data.service_data.metadata.method_count);
88
89 // the request for an incoming RPC must be performed synchronously
90 auto& queue = method_data_.service_data.settings.queue;
91 method_data_.service_data.async_service.template Prepare<CallTraits>(
92 method_data_.method_id, context_, initial_request_, raw_responder_,
93 queue, queue, prepare_.GetTag());
94 }
95
96 void operator()() && {
97 if (!prepare_.Wait()) {
98 // the CompletionQueue is shutting down
99 return;
100 }
101
102 // start a concurrent listener immediately, as advised by gRPC docs
103 ListenAsync(method_data_);
104
105 HandleRpc();
106 }
107
108 static void ListenAsync(const MethodData<GrpcppService, CallTraits>& data) {
109 // The new CallData is constructed synchronously and starts keeping
110 // 'reactor' alive using its 'wait_token_'.
111 engine::CriticalAsyncNoSpan(
112 data.service_data.settings.task_processor,
113 utils::LazyPrvalue([&] { return CallData(data); }))
114 .Detach();
115 }
116
117 private:
118 using InitialRequest = typename CallTraits::InitialRequest;
119 using RawCall = typename CallTraits::RawCall;
120 using Call = typename CallTraits::Call;
121
122 void HandleRpc() {
123 const auto call_name = method_data_.call_name;
124 auto& service = method_data_.service;
125 const auto service_method = method_data_.service_method;
126
127 SetupSpan(span_, context_, call_name);
128 utils::FastScopeGuard destroy_span([&]() noexcept { span_.reset(); });
129
130 ugrpc::impl::RpcStatisticsScope statistics_scope(method_data_.statistics);
131 Call responder(context_, call_name, raw_responder_, statistics_scope);
132
133 try {
134 if constexpr (std::is_same_v<InitialRequest, NoInitialRequest>) {
135 (service.*service_method)(responder);
136 } else {
137 (service.*service_method)(responder, std::move(initial_request_));
138 }
139 } catch (const RpcInterruptedError& ex) {
140 ReportNetworkError(ex, call_name, span_->Get());
141 statistics_scope.OnNetworkError();
142 } catch (const std::exception& ex) {
143 ReportHandlerError(ex, call_name, span_->Get());
144 }
145 }
146
147 // 'wait_token_' must be the first field, because its lifetime keeps
148 // ServiceData alive during server shutdown.
149 const utils::impl::WaitTokenStorage::Token wait_token_;
150
151 MethodData<GrpcppService, CallTraits> method_data_;
152
153 grpc::ServerContext context_{};
154 InitialRequest initial_request_{};
155 RawCall raw_responder_{&context_};
156 AsyncMethodInvocation prepare_{};
157 std::optional<tracing::InPlaceSpan> span_{};
158};

160template <typename GrpcppService>
161class ServiceWorkerImpl final : public ServiceWorker {
162 public:
163 template <typename Service, typename... ServiceMethods>
164 ServiceWorkerImpl(ServiceSettings&& settings,
165 ugrpc::impl::StaticServiceMetadata&& metadata,
166 Service& service, ServiceMethods... service_methods)
167 : service_data_(settings, metadata) {
168 start_ = [this, &service, service_methods...] {
169 std::size_t method_id = 0;
170 (CallData<GrpcppService, CallTraits<ServiceMethods>>::ListenAsync(
171 {service_data_, method_id++, service, service_methods}),
172 ...);
173 };
174 }
175
176 ~ServiceWorkerImpl() override {
177 service_data_.wait_tokens.WaitForAllTokens();
178 }
179
180 grpc::Service& GetService() override { return service_data_.async_service; }
181
182 const ugrpc::impl::StaticServiceMetadata& GetMetadata() const override {
183 return service_data_.metadata;
184 }
185
186 void Start() override { start_(); }
187
188 private:
189 ServiceData<GrpcppService> service_data_;
190 std::function<void()> start_;
191};
