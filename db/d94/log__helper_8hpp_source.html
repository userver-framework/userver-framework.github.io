95 template <typename T>
96 LogHelper& operator<<(const T& value) {
97 // NOLINTNEXTLINE(bugprone-branch-clone)
98 if constexpr (std::is_constructible_v<std::string_view, T>) {
99 // noexcept if the conversion is noexcept
100 *this << std::string_view(value);
101 } else if constexpr (std::is_signed_v<T>) {
102 using LongLong = long long;
103 *this << LongLong{value};
104 } else if constexpr (std::is_unsigned_v<T>) {
105 using UnsignedLongLong = unsigned long long;
106 *this << UnsignedLongLong{value};
107 } else if constexpr (std::is_base_of_v<std::exception, T>) {
108 *this << static_cast<const std::exception&>(value);
109 } else if constexpr (meta::kIsOstreamWritable<T>) {
110 EncodingGuard guard{*this, Encode::kValue};
111 // may throw a non std::exception based exception
112 Stream() << value;
113 } else if constexpr (meta::kIsRange<T> &&
114 !formats::common::kIsFormatValue<T>) {
115 // may throw a non std::exception based exception
116 PutRange(value);
117 } else {
118 static_assert(!sizeof(T),
119 "Please implement logging for your type: "
120 "logging::LogHelper& operator<<(logging::LogHelper& lh, "
121 "const T& value)");
122 }
123
124 return *this;
125 }

269template <typename T>
270void LogHelper::PutRange(const T& range) {
271 static_assert(meta::kIsRange<T>);
272 using std::begin;
273 using std::end;
274
275 constexpr std::string_view kSeparator = ", ";
276 *this << '[';
277
278 bool is_first = true;
279 auto curr = begin(range);
280 const auto end_iter = end(range);
281
282 while (curr != end_iter) {
283 if (IsLimitReached()) {
284 break;
285 }
286 if (is_first) {
287 is_first = false;
288 } else {
289 *this << kSeparator;
290 }
291
292 if constexpr (meta::kIsMap<T>) {
293 PutMapElement(*curr);
294 } else {
295 PutRangeElement(*curr);
296 }
297 ++curr;
298 }
299
300 const auto extra_elements = std::distance(curr, end_iter);
301
302 if (extra_elements != 0) {
303 if (!is_first) {
304 *this << kSeparator;
305 }
306 *this << "..." << extra_elements << " more";
307 }
308
309 *this << ']';
310}
