188template <typename T>
189Value ValueBuilder::DoSerialize(const T& t) {
190 static_assert(
191 formats::common::impl::kHasSerialize<Value, T>,
192 "There is no `Serialize(const T&, formats::serialize::To<yaml::Value>)` "
193 "in namespace of `T` or `formats::serizalize`. "
194 ""
195 "Probably you forgot to include the "
196 "<userver/formats/serialize/common_containers.hpp> or you "
197 "have not provided a `Serialize` function overload.");
198
199 return Serialize(t, formats::serialize::To<Value>());
200}

202template <typename T>
203std::enable_if_t<std::is_integral<T>::value && sizeof(T) <= sizeof(long long),
204 Value>
205Serialize(T value, formats::serialize::To<Value>) {
206 using Type = std::conditional_t<std::is_signed<T>::value, long long,
207 unsigned long long>;
208 return yaml::ValueBuilder(static_cast<Type>(value)).ExtractValue();
209}
