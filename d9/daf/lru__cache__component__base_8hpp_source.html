118template <typename Key, typename Value, typename Hash, typename Equal>
119LruCacheComponent<Key, Value, Hash, Equal>::LruCacheComponent(
120 const components::ComponentConfig& config,
121 const components::ComponentContext& context)
122 : LoggableComponentBase(config, context),
123 name_(components::GetCurrentComponentName(config)),
124 static_config_(config),
125 cache_(std::make_shared<Cache>(static_config_.ways,
126 static_config_.GetWaySize())) {
127 cache_->SetMaxLifetime(static_config_.config.lifetime);
128 cache_->SetBackgroundUpdate(static_config_.config.background_update);
129
130 if (static_config_.use_dynamic_config) {
131 LOG_INFO() << "Dynamic LRU cache config is enabled, subscribing on "
132 "dynamic-config updates, cache="
133 << name_;
134
135 config_subscription_ =
136 impl::FindDynamicConfigSource(context).UpdateAndListen(
137 this, "cache." + name_,
138 &LruCacheComponent<Key, Value, Hash, Equal>::OnConfigUpdate);
139 } else {
140 LOG_INFO() << "Dynamic LRU cache config is disabled, cache=" << name_;
141 }
142
143 statistics_holder_ = impl::FindStatisticsStorage(context).RegisterExtender(
144 "cache." + name_, [this](const auto& /*request*/) {
145 return impl::GetCacheStatisticsAsJson(*cache_);
146 });
147
148 invalidator_holder_.emplace(
149 impl::FindComponentControl(context), *this,
150 &LruCacheComponent<Key, Value, Hash, Equal>::DropCache);
151}
